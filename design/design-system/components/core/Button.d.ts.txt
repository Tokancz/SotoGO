import * as React from 'react';

/**
 * Primary action control for ŠotoGO. Game-y, pressable: the primary/reward/danger
 * variants carry a solid 3D base that collapses on :active.
 *
 * @startingPoint section="Core" subtitle="Pressable game button — primary, reward, secondary, ghost" viewport="700x160"
 */
export interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  /** Visual style. @default "primary" */
  variant?: 'primary' | 'reward' | 'secondary' | 'ghost' | 'danger';
  /** @default "md" */
  size?: 'sm' | 'md' | 'lg';
  /** Stretch to container width. @default false */
  fullWidth?: boolean;
  /** Icon node before the label (e.g. a Lucide <i> or <svg>). */
  leadingIcon?: React.ReactNode;
  /** Icon node after the label. */
  trailingIcon?: React.ReactNode;
  /** Render as a different element, e.g. "a". @default "button" */
  as?: 'button' | 'a';
  children?: React.ReactNode;
}

export function Button(props: ButtonProps): JSX.Element;
